- npm 6.14.8
- @vue/cli 4.5
- vue 2.6.11
- eslint 6.7.2

- plugins: Auto Close Tag

- plugins: indent-rainbow
- plugins: Live Server
- plugins: Korean Language Pack

- plugins: Vue VSCode Snippets
- plugins: Git History
